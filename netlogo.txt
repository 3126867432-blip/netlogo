; ================== utility functions ==================
to-report safe-xy [thing]
  if (is-list? thing and length thing = 2) [ report thing ]
  let loc gis:location-of thing
  if (is-list? loc and length loc = 2) [ report loc ]
  report nobody
end

; ====== scale / distance / shadow sampling / UTCI imputation ======

to-report distance-m [x1 y1 x2 y2]
  ;; convert NetLogo world-coordinate differences to meters
  ;;  update-gis-scale  gis-unit-per-patch-x / y
  if (not is-number? gis-unit-per-patch-x) or (gis-unit-per-patch-x <= 0) [
    update-gis-scale
  ]
  let dx_ (x1 - x2)
  let dy_ (y1 - y2)
  report sqrt ((dx_ * gis-unit-per-patch-x) ^ 2 + (dy_ * gis-unit-per-patch-y) ^ 2)
end

to-report shadow-fraction-between [x1 y1 x2 y2]
  ;; compute the fraction of a segment covered by shadow-buffer patches (0-1)
  ;; daytime: sample shadow-buffered patches; night: return 1
  if not daylight? [ report 1 ]
  if shadow-data = nobody [ report 0 ]

  if (not is-number? shadow-sample-step-m) or (shadow-sample-step-m <= 0) [ set shadow-sample-step-m 5 ]
  if (not is-number? shadow-sample-max-n) or (shadow-sample-max-n < 5) [ set shadow-sample-max-n 25 ]

  let segLenM distance-m x1 y1 x2 y2
  if segLenM <= 0 [ report 0 ]

  let n floor (segLenM / shadow-sample-step-m)
  if n < 1 [ set n 1 ]
  if n > shadow-sample-max-n [ set n shadow-sample-max-n ]

  let dx_ (x2 - x1)
  let dy_ (y2 - y1)

  let shaded 0
  let i 0
  while [ i <= n ] [
    let t (i / n)
    let xs (x1 + dx_ * t)
    let ys (y1 + dy_ * t)
    let p patch xs ys
    if (p != nobody) and ([shadow-buffered?] of p) [
      set shaded shaded + 1
    ]
    set i i + 1
  ]

  report shaded / (n + 1)
end

to impute-utci-values
  ;; purpose: UTCI raster may be NoData on some nodes; direct large penalties break paths
  ;; strategy:
  ;; 1) first impute from adjacent nodes (topological averaging)
  ;; 2) then impute by spatial radius search
  ;; 3) if still missing, fall back to utci-fill-value

  ; compute fallback value
  ifelse any? nodes with [utci-valid?] [
    set utci-fill-value mean [utci-value] of nodes with [utci-valid?]
  ] [
    set utci-fill-value utci-discomfort-threshold
  ]

  ; first pass: neighbor average
  ask nodes with [not utci-valid?] [
    let neigh link-neighbors with [utci-valid?]
    if any? neigh [
      set utci-value mean [utci-value] of neigh
      set utci-valid? true
      set utci-imputed? true
    ]
  ]

  ; second pass: spatial search
  let maxR (ifelse-value (is-number? utci-impute-max-radius-m and utci-impute-max-radius-m > 0) [utci-impute-max-radius-m] [50])
  let stepR (ifelse-value (is-number? utci-impute-step-m and utci-impute-step-m > 0) [utci-impute-step-m] [10])

  if (not is-number? gis-unit-per-patch) or (gis-unit-per-patch <= 0) [ update-gis-scale ]
  let maxRu (maxR / gis-unit-per-patch)
  let stepRu (stepR / gis-unit-per-patch)

  ask nodes with [not utci-valid?] [
    let r stepRu
    while [ (not utci-valid?) and (r <= maxRu) ] [
      let cand nodes in-radius r with [utci-valid?]
      if any? cand [
        let nn min-one-of cand [distance myself]
        set utci-value [utci-value] of nn
        set utci-valid? true
        set utci-imputed? true
      ]
      set r r + stepRu
    ]

    ; final fallback: use mean/threshold to keep the model connected
    if not utci-valid? [
      set utci-value utci-fill-value
      set utci-valid? true
      set utci-imputed? true
    ]
  ]
end

to update-gis-scale
  ;; purpose:
  ;; - use meters as a common scale for distance, time, and buffer radius
  ;; - auto-detect whether roads are lon/lat and convert approximately to meters
  ;;
  ;; outputs:
  ;; - gis-unit-per-patch-x / y:1 world-unit  X/Y
  ;; - gis-unit-per-patch:X/Y ()

  if roads = nobody [ stop ]

  let env gis:envelope-of roads
  let xmin item 0 env
  let xmax item 1 env
  let ymin item 2 env
  let ymax item 3 env

  let x-span (xmax - xmin)
  let y-span (ymax - ymin)

  ;; : envelope ,""
  set coords-are-degrees? false
  if (abs xmin <= 180) and (abs xmax <= 180) and (abs ymin <= 90) and (abs ymax <= 90) [
    set coords-are-degrees? true
  ]

  ;;  GIS span
  let x-span-m x-span
  let y-span-m y-span
  if coords-are-degrees? [
    let lat ((ymin + ymax) / 2)
    ;; :1°≈111320m;1°≈111320*cos(lat)
    let m-per-deg-lat 111320
    let m-per-deg-lon (111320 * cos lat)
    set x-span-m (x-span * m-per-deg-lon)
    set y-span-m (y-span * m-per-deg-lat)
  ]

  ;; meters represented by each world unit (patch)
  set gis-unit-per-patch-x (x-span-m / world-width)
  set gis-unit-per-patch-y (y-span-m / world-height)
  set gis-unit-per-patch ((gis-unit-per-patch-x + gis-unit-per-patch-y) / 2)

  ;; default speed
  if (not is-number? walk-speed-mps) or (walk-speed-mps <= 0) [ set walk-speed-mps 1.5 ]
end


to-report load-dataset-fallback [clip-path orig-path ds-label]
  ;; load the clipped file first; fall back to original if needed
  if file-exists? clip-path [
    report gis:load-dataset clip-path
  ]
  if file-exists? orig-path [
    print (word "Clipped version not found for " ds-label "; using original file instead: " orig-path)
    report gis:load-dataset orig-path
  ]
  user-message (word "Cannot find " ds-label "  file:""\n"
                    clip-path "\nfallback file also not found:\n"
                    orig-path "\n\nPlease check file names/paths, or generate the *_clip.shp and matching sidecar files first.")
  report nobody
end

to update-commuter-od-nodes
  ;; 10-minute resolution:
  ;; - morning: 07:30–09:30  ->
  ;; - evening: 17:00–20:00  ->

  if not any? commuters [ stop ]
  if not any? nodes [
    show "update-commuter-od-nodes: Road network has no nodes; cannot map OD pairs to the network"
    stop
  ]

  let dir "to_work"
  if is-evening-slot? current-time-slot [
    set dir "to_home"
  ]

  let light "night"
  if is-daylight-slot? current-time-slot [
    set light "daylight"
  ]

  ask commuters [
    set trip-dir dir
    set light-class light
    set current-time current-time-slot

    if not ((is-number? metro-x) and (is-number? metro-y) and (is-number? office-x) and (is-number? office-y)) [
      let station-xy safe-xy gis:centroid-of metro-id
      let office-xy  safe-xy gis:centroid-of office-id
      if (station-xy != nobody) and (office-xy != nobody) [
        set metro-x item 0 station-xy
        set metro-y item 1 station-xy
        set office-x item 0 office-xy
        set office-y item 1 office-xy
      ]
    ]

    ifelse (is-number? metro-x) and (is-number? metro-y) and (is-number? office-x) and (is-number? office-y) [
      set od-valid? true

      if trip-dir = "to_work" [
        set start-x metro-x
        set start-y metro-y
        set end-x office-x
        set end-y office-y
      ]
      if trip-dir = "to_home" [
        set start-x office-x
        set start-y office-y
        set end-x metro-x
        set end-y metro-y
      ]
    ] [
      set od-valid? false
      set start-x nobody
      set start-y nobody
      set end-x nobody
      set end-y nobody
    ]

    set commuter-start-node nobody
    set commuter-end-node nobody

    if od-valid? [
      let sx start-x
      let sy start-y
      let ex end-x
      let ey end-y

      let stNode node-from-map station-node-map metro-no
      let ofNode node-from-map office-node-map office-no

      ifelse (stNode != nobody) and (ofNode != nobody) [
        if trip-dir = "to_work" [
          set commuter-start-node stNode
          set commuter-end-node ofNode
        ]
        if trip-dir = "to_home" [
          set commuter-start-node ofNode
          set commuter-end-node stNode
        ]
      ] [
        set commuter-start-node min-one-of nodes [ distancexy sx sy ]
        set commuter-end-node   min-one-of nodes [ distancexy ex ey ]
      ]

      setxy sx sy
      set status "waiting"
      set path-index 0
    ]

    if (commuter-start-node = nobody) or (commuter-end-node = nobody) [
      show (word "Commuter " who " OD node mapping failed (dir=" trip-dir ", start=" commuter-start-node ", end=" commuter-end-node ")")
    ]
  ]
end
to-report node-from-map [mp key]
  ;; get a node from a list shaped like [[key node] ...]; return nobody if not found
  if not is-list? mp [ report nobody ]
  let rec filter [p -> (item 0 p = key)] mp
  if empty? rec [ report nobody ]
  report item 1 first rec
end
to attach-access-node [acc base-road-nodes]
  ;; connect an OD access node acc to the road network by linking it to multiple nearby network nodes
  ;; - base-road-nodes pass a snapshot of base road nodes from before access nodes are created
  if (acc = nobody) or (not any? base-road-nodes) [ stop ]

  let k od-connect-max-n
  if (not is-number? k) or (k <= 0) [ set k 6 ]
  if k > count base-road-nodes [ set k count base-road-nodes ]

  ask acc [
    let targets min-n-of k base-road-nodes [ distance myself ]
    ;; optional radius filter (meters)
    if (is-number? od-connect-radius-m) and (od-connect-radius-m > 0) [
      set targets targets with [ distance-m xcoord ycoord ([xcoord] of myself) ([ycoord] of myself) <= od-connect-radius-m ]
    ]

    ask targets [
      if not link-neighbor? myself [
        create-link-with myself [
          let x1 [xcoord] of end1
          let y1 [ycoord] of end1
          let x2 [xcoord] of end2
          let y2 [ycoord] of end2
          set link-weight distance-m x1 y1 x2 y2
          set color white - 3
          set hidden? true
          set thickness 0.1
        ]
      ]
    ]
  ]
end







to-report time->min [ts]
  let hh read-from-string substring ts 0 2
  let mm read-from-string substring ts 3 5
  report (hh * 60 + mm)
end

to-report min->time [m]
  let hh floor (m / 60)
  let mm (m mod 60)

  let hh-str ""
  let mm-str ""

  ifelse hh < 10
    [ set hh-str (word "0" hh) ]
    [ set hh-str (word hh) ]

  ifelse mm < 10
    [ set mm-str (word "0" mm) ]
    [ set mm-str (word mm) ]

  report (word hh-str ":" mm-str)
end

to-report build-time-slots [start-ts end-ts step-min]
  let out []
  let t time->min start-ts
  let t-end time->min end-ts

  while [ t <= t-end ] [
    set out lput (min->time t) out
    set t (t + step-min)
  ]
  report out
end

to-report is-morning-slot? [ts]
  let t time->min ts
  report (t >= time->min "07:30") and (t <= time->min "09:30")
end

to-report is-evening-slot? [ts]
  let t time->min ts
  report (t >= time->min "17:00") and (t <= time->min "20:00")
end

to-report is-daylight-slot? [ts]
  let t time->min ts
  report (t >= time->min "07:30") and (t <= time->min "19:10")
end


to-report replace-string [str old new]
  let result ""
  let i 0
  while [i < length str] [
    ifelse substring str i (i + length old) = old
    [
      set result (word result new)
      set i i + length old
    ]
    [
      set result (word result substring str i (i + 1))
      set i i + 1
    ]
  ]
  report result
end

to-report valid-feature? [feature]
  ; GIS GIS features are usually not lists, so do not test them with is-list?
  if feature = nobody [ report false ]
  let ok true
  carefully [
    let _ gis:vertex-lists-of feature  ; if geometry vertices can be read, treat the feature as valid
  ] [
    set ok false
  ]
  report ok
end

to init-commuter-heterogeneity
  ask commuters [
    ;; Walking speed heterogeneity: approximately 1.1 to 1.8 m/s
    set personal-walk-speed-mps (1.1 + random-float 0.7)

    ;; Departure-time heterogeneity: 0 to 10 minutes delay within the current slot
    set departure-delay-sec random 601
    set departure-delay-ticks departure-delay-sec

    ;; Heat-tolerance heterogeneity: personal UTCI threshold between 30 and 36
    set personal-utci-threshold (30 + random-float 6)

    ;; Route-preference heterogeneity: distance / heat / shade weights
    let a (0.2 + random-float 0.8)
    let b (0.2 + random-float 0.8)
    let c (0.2 + random-float 0.8)
    let s (a + b + c)

    set pref-dist  (a / s)
    set pref-heat  (b / s)
    set pref-shade (c / s)

    set personal-path-type "unassigned"
  ]
end

to-report choose-personal-path-type [c]
  let m1 path-thermal-metrics ([shortest-path] of c)
  let m2 path-thermal-metrics ([min-utci-path] of c)
  let m3 path-thermal-metrics ([min-shadow-path] of c)

  let d1 item 0 m1
  let d2 item 0 m2
  let d3 item 0 m3

  let u1 item 2 m1
  let u2 item 2 m2
  let u3 item 2 m3

  let s1 item 3 m1
  let s2 item 3 m2
  let s3 item 3 m3

  let th [personal-utci-threshold] of c

  let hot1 max (list 0 (u1 - th))
  let hot2 max (list 0 (u2 - th))
  let hot3 max (list 0 (u3 - th))

  ;; Lower score is better: shorter, cooler, and more shaded
  let score1 (([pref-dist] of c) * d1 +
              ([pref-heat] of c) * hot1 * 500 +
              ([pref-shade] of c) * (1 - s1) * 1000)

  let score2 (([pref-dist] of c) * d2 +
              ([pref-heat] of c) * hot2 * 500 +
              ([pref-shade] of c) * (1 - s2) * 1000)

  let score3 (([pref-dist] of c) * d3 +
              ([pref-heat] of c) * hot3 * 500 +
              ([pref-shade] of c) * (1 - s3) * 1000)

  let best-score min (list score1 score2 score3)

  if best-score = score1 [ report "shortest" ]
  if best-score = score2 [ report "min_utci" ]
  report "min_shadow"
end

to init-active-selection-params

  ; -------- default experiment switch: S0 (all on)--------
  set use-shadow? true
  set use-utci? true
  set scenario-id "S0"
  set detour-max-effective comfort-detour-max-mult

  ; -------- more permissive budgets to make cool corridors easier to choose--------
  ; :active selection"/"
  set comfort-dist-budget-mult 1.30
  set comfort-time-budget-mult 1.30
  set comfort-time-budget-add  600

  ; -------- new: discomfort metric and thresholds for significant improvement--------
  ; thermal discomfort threshold: UTCI above this value accumulates discomfort
  set utci-discomfort-threshold 32
  ; penalty strength and exponent; exponent > 1 emphasizes hotspot links
  set utci-discomfort-alpha 0.60
  set utci-discomfort-power 2
  ; sun penalty: unshaded daytime links are more uncomfortable (off at night)
  set sun-penalty-alpha 0.60
  ;; continuous sun-penalty parameterization: stronger under higher exposure
  set shadow-exposure-gamma 2

  ;; OD:each metro station and office building gets an access node connected to several nearby road-network nodes
  set od-connect-max-n 6
  set od-connect-radius-m 80  ;; meters; <=0 means no radius limit, only nearest N nodes

  ;; solar geometry (optional): if externally computed, set it in update-solar-geometry
  set sun-altitude-deg 0
  set sun-azimuth-deg 0
  ; accept a new route only if discomfort improves by at least these thresholds
  set comfort-discomfort-improve-ratio 0.05   ; reduce by at least 5%
  set comfort-discomfort-improve-abs   0      ; can also set a very small positive number to avoid noise

  ; allow additional detour per 1 C increase in average UTCI, with a cap
  set comfort-heat-detour-per-deg 0.05
  set comfort-detour-max-mult 1.80

  ; older parameter retained to avoid noise / reversals in avg UTCI
  set comfort-utci-improve-eps 0.05

  ; reference value for normalizing exposure integrals
  set comfort-utci-ref 35

  ; multi-candidate weights to improve the chance of cooler-but-not-too-long routes
  ; each item: [wDist wHeat wShade]
  set comfort-weight-sets default-comfort-weight-sets
  set comfort-candidate-count length comfort-weight-sets

  ; daylight flag defaults to false and is updated in load-current-time-data

; ================== active path selection: update multi-candidate weights and store them in links time-weights ==================
; ================== active path selection: update multi-candidate weights and store them in links time-weights ==================
to update-comfort-candidate-weights
  ensure-comfort-weight-sets
  ; requires link-weight / utci-cost / shadow-cost to already be updated on links
  if not any? links [ stop ]

  ; --- defensive check: if comfort-weight-sets is not a list, restore defaults ---
  if (not is-list? comfort-weight-sets) [
    set comfort-weight-sets default-comfort-weight-sets
    set comfort-candidate-count length comfort-weight-sets
  ]

  let Lref median [link-weight] of links
  if (not is-number? Lref) or (Lref <= 0) [ set Lref 1 ]

  let Uref comfort-utci-ref
  if (not is-number? Uref) or (Uref <= 0) [ set Uref 35 ]

  ask links [
    let wlist []
    foreach comfort-weight-sets [ ws ->
      let wDist  item 0 ws
      let wHeat  item 1 ws
      let wShade item 2 ws

      ; normalization:
      ; distance term ~ link-weight / Lref
      ; heat exposure term ~ utci-cost / (Uref * Lref) (utci-cost = UTCI * distance)
      ; unshaded term ~ shadow-cost / Lref (shadow-cost = unshaded fraction * distance)
      let w (wDist  * (link-weight / Lref) +
             wHeat  * (utci-cost / (Uref * Lref)) +
             wShade * (shadow-cost / Lref))
      set wlist lput w wlist
    ]
    set time-weights wlist

    ; for compatibility, combined-weight defaults to candidate 1 (balanced)
    if (is-list? time-weights) and (length time-weights > 1) [
      set combined-weight item 1 time-weights
    ]
  ]
end
; ================== active path selection: choose the coolest candidate within budget ==================
to-report choose-best-comfort-path [start-n end-n base-path]
  ; base-path base-path must already exist; select only candidates that significantly lower discomfort without too much detour
  if (not is-list? base-path) or (length base-path < 2) [
    report (list [] "invalid")
  ]

  let mS path-thermal-metrics base-path
  let distS item 0 mS
  let timeS item 1 mS
  let utciS item 2 mS
  let discS item 4 mS

  if distS <= 0 [ report (list base-path "shortest") ]
  ; if the baseline path has no above-threshold discomfort, no detour is needed
  if discS <= 0 [ report (list base-path "shortest") ]

  ; detour budgets are based on comfort-dist/time-budget-mult and bounded by comfort-detour-max-mult; hotter paths allow more detour
  let heatLevel max (list 0 (utciS - utci-discomfort-threshold))

  let detour-min 1.0
  let detour-max comfort-detour-max-mult
  set detour-max-effective detour-max

  let distMult max (list detour-min (min (list detour-max (comfort-dist-budget-mult + heatLevel * comfort-heat-detour-per-deg))))
  let timeMult max (list detour-min (min (list detour-max (comfort-time-budget-mult + heatLevel * comfort-heat-detour-per-deg))))
  let distBudget distS * distMult
  let timeBudget max (list (timeS * timeMult) (timeS + comfort-time-budget-add))

  ; "significant-improvement threshold: reduce discS by at least a ratio and at least an absolute amount
  let improveNeed max (list (discS * comfort-discomfort-improve-ratio) comfort-discomfort-improve-abs)

  ; candidate types: first discomfort-cost, then utci-cost and the comfort-i sets
  let candidate-types []
  set candidate-types lput "discomfort-cost" candidate-types
  set candidate-types lput "utci-cost" candidate-types

  let i 0
  while [ i < comfort-candidate-count ] [
    set candidate-types lput (word "comfort-" i) candidate-types
    set i i + 1
  ]

  ; "maximum shade"()
  if daylight? [
    set candidate-types lput "shadow-cost" candidate-types
  ]

  let bestPath base-path
  let bestType "shortest"
  let bestDisc discS

  foreach candidate-types [ wt ->
    let p compute-dijkstra start-n end-n wt
    if (is-list? p) and (length p >= 2) [
      let m path-thermal-metrics p
      let d item 0 m
      let tt item 1 m
      let disc item 4 m
      let u item 2 m

      if (d > 0) and (d <= distBudget) and (tt <= timeBudget) [
        ; must meet the significant discomfort-reduction threshold
        if disc <= (discS - improveNeed) [
          ; also avoid a clear worsening in average UTCI
          if u <= (utciS + 0.10) [
            if disc < bestDisc [
              set bestDisc disc
              set bestPath p
              set bestType wt
            ]
          ]
        ]
      ]
    ]
  ]

  report (list bestPath bestType)
end

to load-utci-asc [time-slot]

  ;; : utci-file-map  list,
  if not is-list? utci-file-map [

to-report compute-dijkstra [start-node-param end-node-param weight-type]
  if start-node-param = end-node-param [ report (list start-node-param) ]

  ask nodes [
    set node-dist 1.0e20
    set node-prev nobody
    set visited? false
  ]
  ask start-node-param [ set node-dist 0 ]

  while [ any? nodes with [not visited? and node-dist < 9.9e19] ] [
    let current-node min-one-of (nodes with [not visited?]) [node-dist]
    if current-node = nobody [ report [] ]
    ask current-node [ set visited? true ]

    ask current-node [
      let neighbor-nodes link-neighbors with [not visited?]
      ask neighbor-nodes [
        let edge-weight 0
        if weight-type = "link-weight" [
          set edge-weight [link-weight] of (link-with current-node)
        ]
        if weight-type = "combined-weight" [
          set edge-weight [combined-weight] of (link-with current-node)
        ]
        if weight-type = "shadow-weight" [
          set edge-weight [shadow-weight] of (link-with current-node)
        ]
        if weight-type = "shadow-cost" [
          set edge-weight [shadow-cost] of (link-with current-node)
        ]
        if weight-type = "utci-weight" [
          set edge-weight [utci-weight] of (link-with current-node)
        ]
        if weight-type = "utci-cost" [
          set edge-weight [utci-cost] of (link-with current-node)
        ]
        if weight-type = "discomfort-cost" [
          set edge-weight [discomfort-cost] of (link-with current-node)
        ]
        ; :comfort-N( links  time-weights  N )
        if (length weight-type > 8) and (substring weight-type 0 8 = "comfort-") [
          let idx read-from-string substring weight-type 8 length weight-type
          let tw [time-weights] of (link-with current-node)
          ifelse (is-list? tw) and (idx >= 0) and (idx < length tw) [
            set edge-weight item idx tw
          ] [
            set edge-weight 1.0e20
          ]
        ]

        if node-dist > [node-dist] of current-node + edge-weight [
          set node-dist [node-dist] of current-node + edge-weight
          set node-prev current-node
        ]
      ]
    ]

    if current-node = end-node-param [
      let path-list []
      let current end-node-param
      while [ current != nobody ] [
        set path-list fput current path-list
        set current [node-prev] of current
      ]
      report path-list
    ]
  ]

  report []
end

to build-road-network
  ask nodes [ die ]
  ask links [ die ]

  if roads = nobody [
    user-message "Roads dataset not loaded; cannot build network"
    stop
  ]

  ;;
  if (not is-number? node-merge-tol-m) or (node-merge-tol-m <= 0) [ set node-merge-tol-m 1 ]
  if (not is-number? gis-unit-per-patch) or (gis-unit-per-patch <= 0) [ update-gis-scale ]

  ;; use a meter-based tolerance to quantize vertices and reduce duplicates/disconnections
  let tol (node-merge-tol-m / gis-unit-per-patch)
  if (not is-number? tol) or (tol <= 0) [ set tol 0.0001 ]

  let node-index table:make

  let feats gis:feature-list-of roads
  foreach feats [ f ->
    let vlists gis:vertex-lists-of f

    foreach vlists [ vlist ->
      if is-list? vlist and length vlist > 1 [
        let prev-node nobody

        foreach vlist [ v ->
          let loc nobody
          if is-list? v and length v = 2 [
            set loc v
          ]
          if loc = nobody [
            let tmp gis:location-of v
            if is-list? tmp and length tmp = 2 [
              set loc tmp
            ]
          ]

          if loc != nobody [
            let x0 item 0 loc
            let y0 item 1 loc

            ;; quantized coordinate key
            let qx round (x0 / tol)
            let qy round (y0 / tol)
            let key (word qx "," qy)

            let cur nobody
            if table:has-key? node-index key [
              set cur table:get node-index key
            ]

            if cur = nobody [
              create-nodes 1 [
                set xcoord x0
                set ycoord y0
                setxy xcoord ycoord
                set size 0
                set color white
                set hidden? true
              ]
              ;; the newly created node has the largest who id
              set cur max-one-of nodes [who]
              table:put node-index key cur
            ]

            if (prev-node != nobody) and (cur != nobody) and (cur != prev-node) [
              ask prev-node [
                if not link-neighbor? cur [
                  create-link-with cur [
                    let x1 [xcoord] of end1
                    let y1 [ycoord] of end1
                    let x2 [xcoord] of end2
                    let y2 [ycoord] of end2
                    set link-weight distance-m x1 y1 x2 y2   ;; always use meters
                    set color white - 3
                    set hidden? true
                    set thickness 0.1
                  ]
                ]
              ]
            ]

            set prev-node cur
          ]
        ]
      ]
    ]
  ]

; ================== load data for the current time slot ==================
to load-current-time-data
  let time-string current-time-slot

  ;; clear current shadow data and previous shadow turtles each time slot
  set shadow-data nobody
  ask shadows [ die ]

  ;; decide whether the slot is daylight; daylight sun penalty still applies even if shadow files are missing
  set daylight? false
  if is-list? daylight-time-slots [
    if member? time-string daylight-time-slots [ set daylight? true ]
  ]

  ;; shadow file date is fixed by shadow-date-string in globals
  let shadow-features []
  ;; directory containing shadow shapefiles
  let shadow-base-dir "C:/Users/31268/Desktop/futianCBD/clipped_to_zone/vectors/"
  ;; time-string for example, "07:30" -> "0730"
  let time-tag (word substring time-string 0 2 substring time-string 3 5)
  let shadow-path (word shadow-base-dir "shadow_" shadow-date-string "_" time-tag "_clean_clip.shp")

  ifelse file-exists? shadow-path [
    set shadow-data gis:load-dataset shadow-path
    ;;
    set daylight? true
    show (word "Loaded shadow data: " shadow-path)

    if shadow-data != nobody [
      set shadow-features gis:feature-list-of shadow-data

      ;; do not draw shadow blocks; shadow-data is only used for later calculations
    ]
  ] [
    ;;  file:
    ;; - :""
    ;; - :(daylight?=false)
    if daylight? [
      show (word "  file:(): " shadow-path)
    ]
    if not daylight? [
      show (word " Shadow file not found; processing as night / no-shadow: " shadow-path)
    ]
  ]

  ;; update solar altitude/azimuth (optional, used for continuous sun penalty)
  update-solar-geometry

  ;; load the UTCI ASCII raster for the current time slot
  load-utci-asc current-time-slot

  ;; update scale so buffers/distances are in meters
  if (not is-number? gis-unit-per-patch) or (gis-unit-per-patch <= 0) [ update-gis-scale ]

  ;; check whether the UTCI raster overlaps the roads layer (diagnostic only)
  if current-utci-dataset != nobody [
    let r-env gis:envelope-of roads
    let u-env gis:envelope-of current-utci-dataset
    let rminx item 0 r-env
    let rmaxx item 1 r-env
    let rminy item 2 r-env
    let rmaxy item 3 r-env
    let uminx item 0 u-env
    let umaxx item 1 u-env
    let uminy item 2 u-env
    let umaxy item 3 u-env

    if (rmaxx < uminx) or (umaxx < rminx) or (rmaxy < uminy) or (umaxy < rminy) [
      show " UTCI raster does not overlap the road network. Check CRS/extent and reproject or resample in GIS first."
      show (word "roads env=" r-env)
      show (word "utci  env=" u-env)
      stop
    ]
  ]

  ;; ---- combine building shadow and plant shade into shadow-buffered? ----
  ask patches [ set shadow-buffered? false ]

  if ((not is-number? shadow-buffer-m) or (shadow-buffer-m <= 0)) [
    set shadow-buffer-m 10
  ]
  let r (shadow-buffer-m / gis-unit-per-patch)

  ;; 1) building shadow (time-varying shapefile)
  if daylight? and (shadow-data != nobody) [
    let shadePatches (patches gis:intersecting shadow-data)
    ask shadePatches [
      set shadow-buffered? true
      ask (patches in-radius r) [ set shadow-buffered? true ]
    ]
  ]

  ;; 2) plant shade (binary asc)
  ;; convention: raster values > 0 indicate plant shade / tree shade
  if plant-shadow-loaded? and (plant-shadow-raster != nobody) [
    ask patches [
      let gval gis:raster-sample plant-shadow-raster (list pxcor pycor)
      if (gval != false) and is-number? gval [
        if gval > 0 [
          set shadow-buffered? true
          ask (patches in-radius r) [ set shadow-buffered? true ]
        ]
      ]
    ]
  ]

  ;; update node shadow-value and utci-value
  ask nodes [
    set shadow-value 0

    if daylight? [
      if [shadow-buffered?] of patch-here [
        set shadow-value 1
      ]

      ;; plant shade fallback: directly sample greensp_binary.asc at node coordinates
      if (shadow-value = 0) and plant-shadow-loaded? and (plant-shadow-raster != nobody) [
        let gval gis:raster-sample plant-shadow-raster (list xcoord ycoord)
        if (gval != false) and is-number? gval [
          if gval > 0 [
            set shadow-value 1
          ]
        ]
      ]
    ]

    if not daylight? [
      ;; at night, set shadow-value to 1 to mean no sun exposure penalty
      set shadow-value 1
    ]

    ;; sample the UTCI raster
    set utci-value nobody
    set utci-valid? false
    set utci-imputed? false

    if current-utci-dataset != nobody [
      let sample-value gis:raster-sample current-utci-dataset (list xcoord ycoord)

      ;; accept only numeric raster samples; treat others as nobody
      if (sample-value != false) and is-number? sample-value [
        if sample-value > -9000 [
          set utci-value sample-value
          set utci-valid? true
        ]
      ]
    ]
  ]

  ;; count raw invalid UTCI samples (sampling failure / NoData)
  let invalid0 count nodes with [not utci-valid?]
  if invalid0 > 0 [
    show (word "Number of nodes with invalid UTCI samples (sampling failure / NoData): " invalid0)
  ]

  ;; impute UTCI values to avoid extreme link weights
  impute-utci-values

  ;; update link weights (distance / shadow / heat exposure / discomfort)
  ask links [
    let node1 end1
    let node2 end2

    let x1 [xcoord] of node1
    let y1 [ycoord] of node1
    let x2 [xcoord] of node2
    let y2 [ycoord] of node2

    ;; distance: always use meters
    set link-weight distance-m x1 y1 x2 y2

    ;; average UTCI (or a fallback constant if use-utci? is off)
    let utci1 [utci-value] of node1
    let utci2 [utci-value] of node2
    if not is-number? utci1 [ set utci1 utci-fill-value ]
    if not is-number? utci2 [ set utci2 utci-fill-value ]
    let avg-utci (utci1 + utci2) / 2
    if not use-utci? [ set avg-utci utci-fill-value ]
    set utci-weight avg-utci

    ;; shadow fraction (0-1): sample buffered patches; if use-shadow? is off, treat as fully shaded (=1)
    if use-shadow? [
      if daylight? [
        set shadow-weight shadow-fraction-between x1 y1 x2 y2
      ]
      if not daylight? [
        set shadow-weight 1
      ]
    ]
    if not use-shadow? [ set shadow-weight 1 ]

    ;; cost for routing/comparison: if shadows are off, shadow-cost collapses to shortest path distance
    if not use-shadow? [
      set shadow-cost link-weight
    ]
    if use-shadow? [
      if daylight? [
        set shadow-cost ((1 - shadow-weight) * link-weight)  ; unshaded exposure distance (meters)
      ]
      if not daylight? [
        ;; at night, shadow is not an extra cost; collapse to shortest path distance
        set shadow-cost link-weight
      ]
    ]

    ;; heat exposure: if UTCI is off, utci-cost collapses to shortest path distance
    if use-utci? [
      set utci-cost (avg-utci * link-weight)                 ; heat exposure = UTCI * distance
    ]
    if not use-utci? [ set utci-cost link-weight ]

    ;; thermal discomfort integral and routing weight (emphasizes hotspots)
    if use-utci? [
      let exceed max (list 0 (avg-utci - utci-discomfort-threshold))
      set discomfort-excess ((exceed ^ utci-discomfort-power) * link-weight)

      ;; increase discomfort penalty for unshaded daytime links (only when shadow effects are enabled)
      ;; continuous sun penalty: intensity = sun-penalty-alpha * sun-factor * (exposure ^ shadow-exposure-gamma)
      ;; - exposure = 1 - shadow-weight (0~1),larger under stronger exposure; avoids threshold jumps around shadow=0.49/0.51
      ;; - sun-factor derived from sun-altitude-deg (default daylight=45, night=0)
      if use-shadow? and daylight? [
        let exposure max (list 0 (1 - shadow-weight))
        set discomfort-excess discomfort-excess * (1 + (sun-penalty-alpha * sun-factor * (exposure ^ shadow-exposure-gamma)))
      ]

      ;; discomfort-cost:weight used by dijkstra (distance + discomfort penalty)
      set discomfort-cost (link-weight + utci-discomfort-alpha * discomfort-excess)
    ]
    if not use-utci? [
      set discomfort-excess 0
      set discomfort-cost link-weight
    ]

    ;; combined-weight  comfort  update-comfort-candidate-weights
  ]

  ;; update multi-candidate comfort weights
  update-comfort-candidate-weights

  ;; calculate UTCI range using only directly sampled values; skip if UTCI is off
  if use-utci? [ calculate-utci-range ]
  if not use-utci? [ set min-utci 0 set max-utci 0 ]
  if use-utci? [
    let sampled-utci [utci-value] of nodes with [utci-valid? and (not utci-imputed?)]
    ifelse empty? sampled-utci
    [ show "UTCI range (direct samples only): none" ]
    [ show (word "UTCI range (direct samples only): " (min sampled-utci) "" (max sampled-utci)) ]

    ;; count imputed UTCI values for diagnosis
    let imputedN count nodes with [utci-imputed?]
    if imputedN > 0 [
      show (word "Number of UTCI-imputed/fallback nodes: " imputedN)
    ]
  ]
  if not use-utci? [ show "UTCI disabled: skip UTCI range/imputation diagnostics" ]
end


; ================== calculate UTCI range ==================
to calculate-utci-range
  ;; only measure directly sampled UTCI values for data-quality diagnosis
  let utci-values [utci-value] of nodes with [utci-valid? and (not utci-imputed?)]

  ifelse empty? utci-values [
    set min-utci 0
    set max-utci 0
    show "Warning: no directly sampled UTCI values are available for range calculation"
  ] [
    set min-utci min utci-values
    set max-utci max utci-values
    show (word "UTCI(): " min-utci "" max-utci)
  ]
end



; ================== walking logic ==================
to go
  ;; walking logic:
  ;; -  tick  walk-speed-mps (m/s)  world-unit
  ;; - hit-threshold ""

  if (not is-number? gis-unit-per-patch) or (gis-unit-per-patch <= 0) [ update-gis-scale ]
  let v (ifelse-value (is-number? walk-speed-mps and walk-speed-mps > 0) [walk-speed-mps] [1.5])

  ;; m/s -> world-unit/tick
  let step (v / gis-unit-per-patch)
  let hit-threshold (step * 0.6)

  ask commuters with [status = "walking"] [
    if path-index < length commuter-path [
      let target item path-index commuter-path
      if target != nobody [
        let dist-to-target distance target

        if dist-to-target <= hit-threshold [
          move-to target
          set path-index path-index + 1

          if path-index >= length commuter-path [
            set status "arrived"
          ]
        ]
        if dist-to-target > hit-threshold [
          facexy [xcor] of target [ycor] of target
          forward step
        ]
      ]
    ]
  ]

  tick

  ;; ,
  if all? commuters [status = "arrived"] [
    advance-to-next-time
  ]
end

end

      let total-utci 0

    set travel-time total-distance / v0
      if total-distance > 0 [

    if path-type = "min_shadow" and length min-shadow-path > 0 [

      let total-distance 0
      let total-utci 0
      let total-shadow 0
      let path-length length commuter-path

      if path-length > 1 [
        let node-switch-shadow-stat-first first commuter-path
        foreach but-first commuter-path [ node-switch-shadow-stat-b ->
          if (node-switch-shadow-stat-first != nobody and node-switch-shadow-stat-b != nobody) [

            ]

      ]

      set distance-traveled total-distance
      let v0 personal-walk-speed-mps
    if (not is-number? v0) or (v0 <= 0) [ set v0 walk-speed-mps ]
    set travel-time total-distance / v0
      if total-distance > 0 [

; =====================================================
; compute path metrics: total distance, travel time, average UTCI, average shadow
; path is a list of nodes
; =====================================================
to-report path-thermal-metrics [path]
  ;; extended metrics used to assess significant reduction in thermal discomfort
  ;; returns: [total distance (m) travel time (s) avg UTCI avg shadow discomfort integral peak UTCI hot distance (m)]
  ;; discomfort integral: sum(max(0,UTCI-threshold)^power * distance * daytime unshaded penalty)

  if (not is-list? path) or (length path < 2) [
    report (list 0 0 0 0 0 0 0)
  ]

  ;; speed fallback
  let v (ifelse-value (is-number? walk-speed-mps and walk-speed-mps > 0) [walk-speed-mps] [1.5])
  ;; UTCI fallback value in case utci-fill-value is not yet initialized
  let fill (ifelse-value is-number? utci-fill-value [utci-fill-value] [utci-discomfort-threshold])

  let total-distance 0
  let total-utci 0
  let total-shadow 0

  let total-discomfort 0
  let path-max-utci -1.0e9
  let hot-distance 0

  let a first path
  foreach but-first path [ b ->
    if (a != nobody and b != nobody) [
      let lk ([link-with b] of a)
      if lk != nobody [
        let d [link-weight] of lk
        if not is-number? d [ set d 0 ]
        set total-distance total-distance + d

        let u1 [utci-value] of a
        let u2 [utci-value] of b
        if not is-number? u1 [ set u1 fill ]
        if not is-number? u2 [ set u2 fill ]
        ;; if UTCI is disabled, treat UTCI as a constant fill value
        if not use-utci? [
          set u1 fill
          set u2 fill
        ]
        let uavg (u1 + u2) / 2

        let s [shadow-weight] of lk
        if not is-number? s [ set s 0 ]
        ;; if shadow is disabled, treat every link as fully shaded (=1)
        if not use-shadow? [ set s 1 ]

        set total-utci total-utci + (uavg * d)
        set total-shadow total-shadow + (s * d)

        if uavg > path-max-utci [ set path-max-utci uavg ]
        if use-utci? and (uavg > utci-discomfort-threshold) [ set hot-distance hot-distance + d ]

        let ex 0
        if use-utci? [
          let exceed max (list 0 (uavg - utci-discomfort-threshold))
          set ex (exceed ^ utci-discomfort-power) * d
          ;; ""
           ;; ():keep consistent with update-link-weights
           if use-shadow? and daylight? [
             let exposure max (list 0 (1 - s))
             set ex ex * (1 + (sun-penalty-alpha * sun-factor * (exposure ^ shadow-exposure-gamma)))
           ]
        ]
        set total-discomfort total-discomfort + ex
      ]
    ]
    set a b
  ]

  let avg-utci-exposure 0
  let avg-shadow-exposure 0
  if total-distance > 0 [
    set avg-utci-exposure total-utci / total-distance
    set avg-shadow-exposure total-shadow / total-distance
  ]

  let tt 0
  if total-distance > 0 [ set tt (total-distance / v) ]

  if path-max-utci < -1.0e8 [ set path-max-utci 0 ]

  report (list total-distance tt avg-utci-exposure avg-shadow-exposure total-discomfort path-max-utci hot-distance)
end

to-report path-signature [path]
  ;; output a path signature so Python can compare actual node sequences
  ;; example format: 12-58-77-103
  if (not is-list? path) or empty? path [
    report ""
  ]

  let sig ""
  let first-node? true

  foreach path [ nd ->
    let piece "NA"
    if nd != nobody [
      set piece [who] of nd
    ]

    ifelse first-node? [
      set sig (word piece)
      set first-node? false
    ] [
      set sig (word sig "-" piece)
    ]
  ]

  report sig
end

to-report heat-class-from-metrics [m]
  ;; classify paths as OK / HOT based on discomfort and threshold
  ;; m = path-thermal-metrics outputs: [dist time avgUTCI avgShade discomfort peakUTCI hotDist]
  if (not is-list? m) or (length m < 7) [ report "NA" ]

  let avg item 2 m
  let peak item 5 m
  let dis item 4 m

  if (is-number? dis) and (dis > 0) [ report "HOT" ]
  if (is-number? peak) and (peak > utci-discomfort-threshold) [ report "HOT" ]
  if (is-number? avg) and (avg > utci-discomfort-threshold) [ report "HOT" ]

  report "OK"
end

to clip-network-to-zone
  ;; zone is the site boundary; nodes and links outside it are removed
  if zone-data = nobody [
    show ":zone-data not loaded; skipping clipping"
    stop
  ]

  let zone-features gis:feature-list-of zone-data
  if empty? zone-features [
    show ":zone-data has no features; skipping clipping"
    stop
  ]
  let zone-poly first zone-features

  ;; 1) remove nodes outside the zone
  ask nodes [
    let pt (list xcoord ycoord)
    if not gis:intersects? zone-poly pt [
      die
    ]
  ]

  ;; 2) remove links whose endpoints are missing or no longer in the node set
  ask links [
    if (end1 = nobody) or (end2 = nobody) [ die ]
    if (not member? end1 nodes) or (not member? end2 nodes) [ die ]
  ]

  show (word "After clipping: nodes=" count nodes ", =" count links)
end

; =====================================================
; recompute each commuter's three paths when time-dependent weights change
; =====================================================
to recompute-commuter-3paths
  ;; :
  ;; 1) (link-weight)
  ;; 2) active selection:
  ;; 3) maximum shade(;shortest route)

  ;; key robustness improvements:
  ;; - if start/end node references are invalid, first remap using cached OD coordinates
  ;; - when falling back to centroids, respect the current trip direction
  ;; - avoid all-zero batch exports caused by invalid GIS feature handles

  ask commuters [
    ;; 1) self-healing: if nodes are lost, try to remap using cached coordinates
    if (commuter-start-node = nobody) or (commuter-end-node = nobody) [
      if od-valid? [
        set commuter-start-node min-one-of nodes [ distancexy ([start-x] of myself) ([start-y] of myself) ]
        set commuter-end-node   min-one-of nodes [ distancexy ([end-x] of myself)   ([end-y] of myself) ]
      ]
    ]

    ;; 2) fallback: if cached mapping fails, derive centroids from GIS features and rebuild OD by trip direction
    if (commuter-start-node = nobody) or (commuter-end-node = nobody) [
      let station-xy safe-xy gis:centroid-of metro-id
      let office-xy  safe-xy gis:centroid-of office-id

      if (station-xy != nobody) and (office-xy != nobody) [
        set metro-x item 0 station-xy
        set metro-y item 1 station-xy
        set office-x item 0 office-xy
        set office-y item 1 office-xy

        if trip-dir = "to_work" [
          set start-x metro-x
          set start-y metro-y
          set end-x   office-x
          set end-y   office-y
        ]
        if trip-dir = "to_home" [
          set start-x office-x
          set start-y office-y
          set end-x   metro-x
          set end-y   metro-y
        ]

        set od-valid? true
        set commuter-start-node min-one-of nodes [ distancexy ([start-x] of myself) ([start-y] of myself) ]
        set commuter-end-node   min-one-of nodes [ distancexy ([end-x] of myself)   ([end-y] of myself) ]
      ]
    ]

    ;; 3) if still invalid, leave paths empty (CSV will output zeros)
    ifelse (commuter-start-node = nobody) or (commuter-end-node = nobody) [
      set shortest-path []
      set min-utci-path []
      set min-shadow-path []
      set current-path-type "invalid"
    ] [
      ;; (1) shortest-distance path
      set shortest-path compute-dijkstra commuter-start-node commuter-end-node "link-weight"

      ;; (2) active selection:
      let chosen choose-best-comfort-path commuter-start-node commuter-end-node shortest-path
      set min-utci-path item 0 chosen
      set current-path-type item 1 chosen

      ;; (3) maximum shade:;( 0 )
      ifelse daylight? [
        set min-shadow-path compute-dijkstra commuter-start-node commuter-end-node "shadow-cost"
      ] [
        set min-shadow-path shortest-path
      ]
    ]
  ]
end

to compute-components
  ;; assign each node a connected-component id based on link-neighbors
  ask nodes [ set comp-id -1 ]
  let cid 0
  while [ any? nodes with [ comp-id = -1 ] ] [
    let seed one-of nodes with [ comp-id = -1 ]
    ask seed [ set comp-id cid ]
    let frontier (turtle-set seed)
    while [ any? frontier ] [
      let new-neigh (turtle-set [ link-neighbors ] of frontier) with [ comp-id = -1 ]
      ifelse any? new-neigh [
        ask new-neigh [ set comp-id cid ]
        set frontier new-neigh
      ] [
        set frontier no-turtles
      ]
    ]
    set cid cid + 1
  ]
end


to connect-required-components
  ;; bridge connected components that contain station/office access nodes so any OD pair is reachable
  let required-nodes (turtle-set (map [pair -> item 1 pair] station-node-map)
                                (map [pair -> item 1 pair] office-node-map))
  if not any? required-nodes [ stop ]

  let guard 0
  while [guard < 50] [
    set guard guard + 1
    compute-components

    let req-comps remove-duplicates [comp-id] of required-nodes
    if length req-comps <= 1 [ stop ]

    let main-comp first req-comps
    let main-set nodes with [comp-id = main-comp]
    let other-comps but-first req-comps

    foreach other-comps [ c ->
      let other-set nodes with [comp-id = c]
      let bestA nobody
      let bestB nobody
      let bestD 1e18

      ask other-set [
        let a self
        let b min-one-of main-set [ distancexy [xcor] of a [ycor] of a ]
        if b != nobody [
          let d distance-m ([xcor] of a) ([ycor] of a) ([xcor] of b) ([ycor] of b)
          if d < bestD [
            set bestD d
            set bestA a
            set bestB b
          ]
        ]
      ]

      if (bestA != nobody) and (bestB != nobody) [
        ask bestA [
          create-link-with bestB [
            set link-weight bestD
            set shadow-cost bestD
            set utci-cost bestD
            set discomfort-cost bestD
            set combined-weight bestD
            set time-weights []
          ]
        ]
      ]
    ]
  ]
end


; ================== scenario switches for counterfactual / ablation experiments ==================
to set-experiment-scenario [sid]
  ;; S0: all on (baseline scenario)
  ;; S1: shadow off (UTCI only)
  ;; S2: UTCI off (shadow only)
  ;; S3: all off (geometry / distance only)
  set scenario-id sid
  if sid = "S0" [ set use-shadow? true  set use-utci? true  ]
  if sid = "S1" [ set use-shadow? false set use-utci? true  ]
  if sid = "S2" [ set use-shadow? true  set use-utci? false ]
  if sid = "S3" [ set use-shadow? false set use-utci? false ]
end

; export all S0-S3 scenarios across all 10-minute slots
to export-all-scenarios-7-times-to-desktop
  init-active-selection-params
  ensure-comfort-weight-sets
  if not is-list? time-slots [ setup-time-slots ]

  let scenarios ["S0""S1""S2""S3"]
  foreach scenarios [ sid ->
    set-experiment-scenario sid
    show (word "Starting scenario: " sid " (use-shadow?=" use-shadow? ", use-utci?=" use-utci? ")")
    foreach time-slots [ tt ->
      set current-time-slot tt
      show (word "Processing time slot: " tt)
      ;; (+UTCI); use-shadow?/use-utci?
      load-current-time-data
      recompute-paths-for-current-time
      export-3paths-csv-to-desktop
    ]
  ]
end
